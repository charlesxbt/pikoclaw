// npm install x402-express
import { paymentMiddleware } from 'x402-express'
import { facilitator } from 'x402/facilitators'

const wallet = "0xYourWalletAddress"

app.use(paymentMiddleware(
  wallet,
  {
    "GET /skills/browse-agent": {
      price: "$0.04",
      network: "base",
    },
    "GET /skills/code-review": {
      price: "$0.10",
      network: "base",
    },
  },
  { facilitator: facilitator("base") }
))

// Routes auto-protected by x402
app.get("/skills/:id",
  (req, res) => {
    res.json({ skill: req.params.id, access: "granted" })
  }
)
